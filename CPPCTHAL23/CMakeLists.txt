cmake_minimum_required(VERSION 3.22)

set(CPPCTHAL23_STANDALONE OFF)
if(CMAKE_SOURCE_DIR STREQUAL CMAKE_CURRENT_SOURCE_DIR)
  set(CPPCTHAL23_STANDALONE ON)
  if(NOT CMAKE_TOOLCHAIN_FILE)
    set(
      CMAKE_TOOLCHAIN_FILE
      "${CMAKE_CURRENT_LIST_DIR}/../cmake/toolchains/arm-none-eabi-gcc.cmake"
      CACHE FILEPATH "CMake toolchain file"
    )
  endif()
  project(CPPCTHAL23 LANGUAGES C CXX ASM)
  set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
endif()

include("${CMAKE_CURRENT_LIST_DIR}/../cmake/CompileTimeHal.cmake")
cthal_validate_toolchain()

set(CPPCTHAL23_COMMON_SOURCES
  "${CMAKE_CURRENT_LIST_DIR}/Src/syscalls.c"
  "${CMAKE_CURRENT_LIST_DIR}/Src/sysmem.c"
)
set(CPPCTHAL23_STARTUP
  "${CMAKE_CURRENT_LIST_DIR}/Src/startup_stm32l433xx.S"
)
set(CPPCTHAL23_LINKER_SCRIPT
  "${CTHAL_REPOSITORY_ROOT}/CPPCTHAL/stm32l433xc_flash.ld"
)

cthal_add_firmware(CPPCTHAL23
  MODE STM32CUBE
  MAIN_SOURCE "${CMAKE_CURRENT_LIST_DIR}/Src/main.cpp"
  STARTUP_SOURCE "${CPPCTHAL23_STARTUP}"
  LINKER_SCRIPT "${CPPCTHAL23_LINKER_SCRIPT}"
  CXX_STANDARD 23
  SOURCES ${CPPCTHAL23_COMMON_SOURCES}
  INCLUDE_DIRS "${CMAKE_CURRENT_LIST_DIR}"
)

if(NOT CPPCTHAL23_STANDALONE)
  cthal_add_benchmarks(
    TARGET_PREFIX cppcthal23
    SOURCE_BASENAME cppcthal23.cpp
    STARTUP_SOURCE "${CPPCTHAL23_STARTUP}"
    LINKER_SCRIPT "${CPPCTHAL23_LINKER_SCRIPT}"
    CXX_STANDARD 23
    SOURCES ${CPPCTHAL23_COMMON_SOURCES}
    INCLUDE_DIRS "${CMAKE_CURRENT_LIST_DIR}"
  )
endif()
