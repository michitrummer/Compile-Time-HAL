cmake_minimum_required(VERSION 3.22)

set(CUBEMXLL_STANDALONE OFF)
if(CMAKE_SOURCE_DIR STREQUAL CMAKE_CURRENT_SOURCE_DIR)
  set(CUBEMXLL_STANDALONE ON)
  if(NOT CMAKE_TOOLCHAIN_FILE)
    set(
      CMAKE_TOOLCHAIN_FILE
      "${CMAKE_CURRENT_LIST_DIR}/../cmake/toolchains/arm-none-eabi-gcc.cmake"
      CACHE FILEPATH "CMake toolchain file"
    )
  endif()
  project(CubeMXLL LANGUAGES C ASM)
  set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
endif()

include("${CMAKE_CURRENT_LIST_DIR}/../cmake/CompileTimeHal.cmake")
cthal_validate_toolchain()

set(CUBEMXLL_COMMON_SOURCES
  "${CMAKE_CURRENT_LIST_DIR}/Src/system_stm32l4xx.c"
  "${CMAKE_CURRENT_LIST_DIR}/Drivers/STM32L4xx_HAL_Driver/Src/stm32l4xx_ll_gpio.c"
)
set(CUBEMXLL_INCLUDE_DIRS
  "${CMAKE_CURRENT_LIST_DIR}"
  "${CMAKE_CURRENT_LIST_DIR}/Inc"
  "${CMAKE_CURRENT_LIST_DIR}/Drivers/CMSIS/Include"
  "${CMAKE_CURRENT_LIST_DIR}/Drivers/CMSIS/Device/ST/STM32L4xx/Include"
  "${CMAKE_CURRENT_LIST_DIR}/Drivers/STM32L4xx_HAL_Driver/Inc"
)
set(CUBEMXLL_STARTUP
  "${CMAKE_CURRENT_LIST_DIR}/Startup/startup_stm32l433rctxp.s"
)
set(CUBEMXLL_LINKER_SCRIPT
  "${CTHAL_REPOSITORY_ROOT}/CPPCTHAL/stm32l433xc_flash.ld"
)

cthal_add_firmware(CubeMXLL
  MODE STM32CUBE
  MAIN_SOURCE "${CMAKE_CURRENT_LIST_DIR}/Src/main.c"
  STARTUP_SOURCE "${CUBEMXLL_STARTUP}"
  LINKER_SCRIPT "${CUBEMXLL_LINKER_SCRIPT}"
  SOURCES ${CUBEMXLL_COMMON_SOURCES}
  INCLUDE_DIRS ${CUBEMXLL_INCLUDE_DIRS}
  DEFINES USE_FULL_LL_DRIVER
)

if(NOT CUBEMXLL_STANDALONE)
  cthal_add_benchmarks(
    TARGET_PREFIX cubemxll
    SOURCE_BASENAME cubemx-ll.c
    STARTUP_SOURCE "${CUBEMXLL_STARTUP}"
    LINKER_SCRIPT "${CUBEMXLL_LINKER_SCRIPT}"
    SOURCES ${CUBEMXLL_COMMON_SOURCES}
    INCLUDE_DIRS ${CUBEMXLL_INCLUDE_DIRS}
    DEFINES USE_FULL_LL_DRIVER
  )
endif()
